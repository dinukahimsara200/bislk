Schema Queries — schema.sql
Q1 Create Database DDL 
CREATE DATABASE IF NOT EXISTS bislk_db
  CHARACTER SET utf8mb4
  COLLATE utf8mb4_unicode_ci;

USE bislk_db;
What it does
Creates the bislk_db database with full Unicode support and selects it for use. IF NOT EXISTS prevents an error if the database is already there.
Q2 Create Table: users DDL 
CREATE TABLE users (
    user_id       INT             NOT NULL AUTO_INCREMENT,
    full_name     VARCHAR(100)    NOT NULL,
    email         VARCHAR(150)    NOT NULL,
    password_hash VARCHAR(255)    NOT NULL,
    phone         VARCHAR(20)     DEFAULT NULL,
    city          VARCHAR(100)    DEFAULT NULL,
    latitude      DECIMAL(9,7)    DEFAULT NULL,
    longitude     DECIMAL(10,7)   DEFAULT NULL,
    joined_at     DATETIME        NOT NULL DEFAULT CURRENT_TIMESTAMP,

    PRIMARY KEY (user_id),
    UNIQUE KEY uq_users_email (email)
) ENGINE=InnoDB;
What it does
Creates the users table. Includes city and GPS coordinates to support the nearby-business search feature. Passwords are stored as bcrypt hashes.

Q5 Create Table: businesses DDL 
CREATE TABLE businesses (
    business_id   INT             NOT NULL AUTO_INCREMENT,
    name          VARCHAR(200)    NOT NULL,
    address       TEXT            NOT NULL,
    city          VARCHAR(100)    NOT NULL,
    latitude      DECIMAL(10,7)   DEFAULT NULL,
    longitude     DECIMAL(10,7)   DEFAULT NULL,
    phone         VARCHAR(20)     DEFAULT NULL,
    email         VARCHAR(150)    DEFAULT NULL,
    website       VARCHAR(500)    DEFAULT NULL,
    description   TEXT            DEFAULT NULL,
    category_id   INT             NOT NULL,
    owner_id      INT             DEFAULT NULL,
    is_verified   TINYINT(1)      NOT NULL DEFAULT 0,
    created_at    DATETIME        NOT NULL DEFAULT CURRENT_TIMESTAMP,

    PRIMARY KEY (business_id),
    KEY idx_businesses_category  (category_id),
    KEY idx_businesses_owner     (owner_id),
    KEY idx_businesses_city      (city),
    KEY idx_businesses_location  (latitude, longitude),

    CONSTRAINT fk_businesses_category
        FOREIGN KEY (category_id) REFERENCES categories (category_id)
        ON UPDATE CASCADE ON DELETE RESTRICT,

    CONSTRAINT fk_businesses_owner
        FOREIGN KEY (owner_id) REFERENCES owners (owner_id)
        ON UPDATE CASCADE ON DELETE SET NULL
) ENGINE=InnoDB;
What it does
The main table — every business listing lives here. Stores name, location, contact details, and links to a category. Lat/lon coordinates enable the nearby search.

Stores photos for each business. Each row links to a business and one image is flagged as the primary cover photo shown in cards and search results.
Q7 Create Table: reviews DDL 
CREATE TABLE reviews (
    review_id     INT             NOT NULL AUTO_INCREMENT,
    user_id       INT             NOT NULL,
    business_id   INT             NOT NULL,
    rating        TINYINT         NOT NULL,
    title         VARCHAR(200)    DEFAULT NULL,
    body          TEXT            DEFAULT NULL,
    created_at    DATETIME        NOT NULL DEFAULT CURRENT_TIMESTAMP,

    PRIMARY KEY (review_id),
    UNIQUE KEY uq_reviews_user_business (user_id, business_id),
    KEY idx_reviews_business (business_id),

    CONSTRAINT chk_reviews_rating
        CHECK (rating BETWEEN 1 AND 5),

    CONSTRAINT fk_reviews_user
        FOREIGN KEY (user_id) REFERENCES users (user_id)
        ON UPDATE CASCADE ON DELETE CASCADE,

    CONSTRAINT fk_reviews_business
        FOREIGN KEY (business_id) REFERENCES businesses (business_id)
        ON UPDATE CASCADE ON DELETE CASCADE
) ENGINE=InnoDB;
What it does
Stores reviews and star ratings (1–5). A user can review each business only once, enforced by a unique constraint on (user_id, business_id).

Q11 View: vw_business_ratings VIEW 
CREATE OR REPLACE VIEW vw_business_ratings AS
SELECT
    b.business_id,
    b.name,
    b.city,
    c.name          AS category,
    COUNT(r.review_id)          AS review_count,
    ROUND(AVG(r.rating), 1)     AS avg_rating
FROM businesses b
JOIN categories c  ON b.category_id = c.category_id
LEFT JOIN reviews r ON b.business_id = r.business_id
GROUP BY b.business_id, b.name, b.city, c.name;
What it does
A saved view that pre-calculates the average rating and review count for every business. Other queries use it to avoid repeating the same AVG/COUNT logic.
Q12 View: vw_business_detail VIEW 
CREATE OR REPLACE VIEW vw_business_detail AS
SELECT
    b.*,
    c.name          AS category_name,
    c.slug          AS category_slug,
    c.icon          AS category_icon,
    o.full_name     AS owner_name,
    img.image_url   AS primary_image,
    vbr.avg_rating,
    vbr.review_count
FROM businesses b
JOIN  categories c  ON b.category_id = c.category_id
LEFT JOIN owners o  ON b.owner_id    = o.owner_id
LEFT JOIN business_images img
    ON b.business_id = img.business_id AND img.is_primary = 1
LEFT JOIN vw_business_ratings vbr
    ON b.business_id = vbr.business_id;
What it does
A view that combines a business with its category details, primary image URL, and rating stats in one place. Used as a convenient single source for the detail page.

SELECT * FROM vw_business_ratings

ORDER BY avg_rating DESC, review_count DESC

AQ4 Search Businesses (Dynamic Filters) DQL 
SELECT b.business_id, b.name, b.city, b.address, b.phone,
       c.name AS category, c.icon AS category_icon,
       COALESCE(vbr.avg_rating, 0)    AS avg_rating,
       COALESCE(vbr.review_count, 0)  AS review_count,
       img.image_url AS primary_image
FROM businesses b
JOIN categories c ON b.category_id = c.category_id
LEFT JOIN vw_business_ratings vbr ON b.business_id = vbr.business_id
LEFT JOIN business_images img
       ON b.business_id = img.business_id AND img.is_primary = 1
-- WHERE clause built dynamically in Python:
-- (b.name LIKE %q% OR b.description LIKE %q% OR b.address LIKE %q%)
-- AND b.category_id = %category_id%
-- AND b.city LIKE %city%
ORDER BY avg_rating DESC, b.name ASC;
What it does
Powers the search page. Filters by keyword, category, and city — only the conditions the user filled in are added to the WHERE clause.
AQ5 List Distinct Cities (Search Filter) DQL

AQ13 Nearby Businesses — Haversine Formula DQL 
SELECT
    b.business_id, b.name, b.city, b.address,
    c.name              AS category,
    c.slug              AS category_slug,
    c.icon              AS category_icon,
    COALESCE(vbr.avg_rating,   0) AS avg_rating,
    COALESCE(vbr.review_count, 0) AS review_count,
    img.image_url                 AS primary_image,
    ROUND(
      6371 * 2 * ASIN(SQRT(
        POWER(SIN(RADIANS(b.latitude  - %s) / 2), 2) +
        COS(RADIANS(%s)) * COS(RADIANS(b.latitude))  *
        POWER(SIN(RADIANS(b.longitude - %s) / 2), 2)
      )), 2
    ) AS distance_km
FROM businesses b
JOIN  categories c   ON b.category_id  = c.category_id
LEFT JOIN vw_business_ratings vbr ON b.business_id  = vbr.business_id
LEFT JOIN business_images img
                     ON b.business_id  = img.business_id
                    AND img.is_primary = 1
WHERE b.latitude  IS NOT NULL
  AND b.longitude IS NOT NULL
HAVING distance_km <= %s
ORDER BY distance_km ASC
LIMIT 20;
What it does
Finds businesses within a given distance (km) using the Haversine formula on GPS coordinates. Returns up to 20 results, sorted nearest first.

AQ20 Report 1 — Top Rated per Category DQL 
SELECT
    b.business_id,
    c.name              AS category,
    b.name              AS business,
    b.city,
    ROUND(AVG(r.rating), 1)  AS avg_rating,
    COUNT(r.review_id)       AS review_count
FROM reviews r
JOIN businesses b  ON r.business_id = b.business_id
JOIN categories c  ON b.category_id = c.category_id
GROUP BY b.business_id, c.name, b.name, b.city
ORDER BY c.name ASC, avg_rating DESC;
What it does
Lists every reviewed business with its average rating and review count, grouped by category. Powers the main table in Report 1.
AQ21 Chart Data — Average Rating per Category DQL 
SELECT c.name AS category, ROUND(AVG(r.rating), 2) AS avg_rating
FROM reviews r
JOIN businesses b ON r.business_id = b.business_id
JOIN categories c ON b.category_id = c.category_id
GROUP BY c.name
ORDER BY avg_rating DESC;
What it does
Calculates the average rating for each category across all businesses. The results feed the bar chart on the Reports page.
AQ22 Report 2 — Most Reviewed Businesses DQL 
SELECT b.name, b.city, c.name AS category,
       COUNT(r.review_id)       AS review_count,
       ROUND(AVG(r.rating), 1)  AS avg_rating
FROM reviews r
JOIN businesses b ON r.business_id = b.business_id
JOIN categories c ON b.category_id = c.category_id
GROUP BY b.business_id, b.name, b.city, c.name
ORDER BY review_count DESC
LIMIT 10;
What it does
Finds the 10 businesses with the most reviews. Shows both review count and average rating so popularity and quality can be compared.
AQ23 Monthly Review Trend (24-Month Window) DQL 
SELECT
    DATE_FORMAT(created_at, '%Y-%m') AS month,
    COUNT(*) AS review_count
FROM reviews
WHERE created_at >= DATE_SUB(NOW(), INTERVAL 24 MONTH)
GROUP BY month
ORDER BY month ASC;
What it does
Counts reviews per month over a rolling 24-month window. The results feed the trend line chart on the Reports page.
